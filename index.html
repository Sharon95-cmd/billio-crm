-- ============================================================ -- BILLIO SUIVI v2 - Mise à jour avec authentification -- ============================================================ -- À exécuter UNIQUEMENT si vous activez l'authentification. -- Cela protège la base : seuls les utilisateurs connectés -- (avec email/mot de passe) peuvent lire/écrire les données. -- -- Copiez tout ce fichier dans Supabase → SQL Editor → New query -- Puis cliquez sur "Run" -- ============================================================ -- 1. Supprimer les anciennes politiques (qui autorisaient tout le monde) DROP POLICY IF EXISTS "Lecture autorisée pour tous" ON crm_data; DROP POLICY IF EXISTS "Insertion autorisée pour tous" ON crm_data; DROP POLICY IF EXISTS "Mise à jour autorisée pour tous" ON crm_data; -- 2. Nouvelles politiques : seuls les utilisateurs authentifiés ont accès CREATE POLICY "Lecture pour utilisateurs connectés" ON crm_data FOR SELECT TO authenticated USING (true); CREATE POLICY "Écriture pour utilisateurs connectés" ON crm_data FOR INSERT TO authenticated WITH CHECK (true); CREATE POLICY "Mise à jour pour utilisateurs connectés" ON crm_data FOR UPDATE TO authenticated USING (true) WITH CHECK (true); -- ============================================================
